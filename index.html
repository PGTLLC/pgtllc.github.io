    <script>
        const slider = document.getElementById('volumeSlider');
        const volumeDisplay = document.getElementById('volumeDisplay');
        const legacyCostSpan = document.getElementById('legacyCost');
        const ourCostSpan = document.getElementById('ourCost');
        const monthlySavingsSpan = document.getElementById('monthlySavings');
        const annualSavingsSpan = document.getElementById('annualSavings');

        function updateCalculator() {
            let volume = parseFloat(slider.value);
            let legacyRate = 0.030;   // 3.0% current
            let ourRate = 0.020;       // 2.0% offered
            let legacyMonthly = Math.round(volume * legacyRate);
            let ourMonthly = Math.round(volume * ourRate);
            let monthlySave = legacyMonthly - ourMonthly;
            let annualSave = monthlySave * 12;

            volumeDisplay.innerText = '$' + volume.toLocaleString();
            legacyCostSpan.innerText = '$' + legacyMonthly.toLocaleString();
            ourCostSpan.innerText = '$' + ourMonthly.toLocaleString();
            monthlySavingsSpan.innerText = '$' + monthlySave.toLocaleString();
            annualSavingsSpan.innerText = '$' + annualSave.toLocaleString();
        }

        slider.addEventListener('input', updateCalculator);
        updateCalculator();
